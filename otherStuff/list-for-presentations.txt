#include <stdio.h>

main(){
       	int n;
       	int zeroCheck = 0;
       	while(zeroCheck == 0){
       		scanf("%d",&n);
       		if(n == 0){zeroCheck = 1;}
       		n = fibonacci(n);
       		printf("%d\n", n);
       	}
}
int fibonacci(int n){
       	if(n == 0){return 0;}
       	if(n == 1){return 1;}
       	return fibonacci(n-1) + fibonacci(n-2);
}
